```c++
#include <iostream>
#include <vector>
#include <string>
#include <algorithm>
using namespace std;

int main()
{
    ios::sync_with_stdio(false);
    cin.tie(NULL);
    cout.tie(NULL);
    
    int M, N;
    cin >> M >> N;
    
    for (int i = M; i <= N; i++){
        int count = 0;
        for (int j = 1; j <= N; j++){
            if (i % j == 0){
                count++;
            }
            
            if (count > 2){
                break;
            }
        }
        
        if (count == 2){
            cout << i << '\n';
        }
    }
    
    return 0;
}
```

```c++
#include <iostream>
#include <vector>
#include <cmath>
using namespace std;

int main()
{
    ios::sync_with_stdio(false);
    cin.tie(NULL);
    cout.tie(NULL);
    
    int M, N;
    cin >> M >> N;
    
    vector<int> prime(N + 1, 0);
    
    for (int i = 2; i <= N; i++){
        prime[i] = i;
    }
    
    for (int i = 2; i <= sqrt(N); i++){
        if (prime[i] == 0){
            continue;
        }
        
        for (int j = i + i; j <= N; j = j + i){
            prime[j] = 0;
        }
    }
    
    for (int i = M; i <= N; i++){
        if (prime[i] != 0){
            cout << prime[i] << '\n';
        }
    }
    
    return 0;
}
```
